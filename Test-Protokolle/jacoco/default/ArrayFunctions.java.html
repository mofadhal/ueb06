public class ArrayFunctions {
	private ArrayFunctions() {
	}
	
	public static double quadratischerMittelwert(final double[] array) {
		if (array == null || array.length == 0) {
			throw new IllegalArgumentException();
		}
		
		double quadraticAverage = 0.0;
		
		for(final double number: array) {
			quadraticAverage = quadraticAverage + number * number;
		}
		
		quadraticAverage = quadraticAverage / array.length;
		quadraticAverage = Math.sqrt(quadraticAverage);
				
		return quadraticAverage;
	}

	public static double groessterAbstandQMW(final double[] array) {
		if (array == null || array.length == 0) {
			throw new IllegalArgumentException();
		}
		
		final double quadraticAverage = quadratischerMittelwert(array);
		double biggestDistance = 0;
		double numberWithBiggestDistance = 0;
		
		for (final double number: array) {
			final double distance = Math.abs(quadraticAverage - number);
			if(distance > biggestDistance) {
				biggestDistance = distance;
				numberWithBiggestDistance = number;
			}
		}
		
		return numberWithBiggestDistance;
	}

	public static void insertionSort(final int[] array) {
		if (array == null || array.length == 0) {
			throw new IllegalArgumentException();
		}
		
		int temp;
		for (int i = 1; i < array.length; i++) {
			temp = array[i];
			int j = i;
			while (j > 0 && array[j - 1] > temp) {
				array[j] = array[j - 1];
				j--;
			}
			array[j] = temp;
		}
	}
}
